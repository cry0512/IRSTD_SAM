GPU=0; SIZE=256; BS=4; EPOCHS=1000; MODEL="vitt"; KP=1; TRAIN_DIR="50_50"; MASK_SFX=""; \
DATASET=$(basename "$ROOT"); SPLIT_TAG="_split-${TRAIN_DIR}"; SFX_TAG=""; \
EXP="${DATASET}_model-${MODEL}_size${SIZE}_HQ_RadialGate_SCPreproc_TUNE_bin8_edge07_bs${BS}_e${EPOCHS}_kp${KP}${SFX_TAG}${SPLIT_TAG}"; \
TS=$(date +'%Y%m%d%H%M%S'); LOG_ROOT="outputs_sam_sirst_logs/${EXP}${TS}"; mkdir -p "$LOG_ROOT"; \
CUDA_VISIBLE_DEVICES=${GPU} PYTHONUNBUFFERED=1 nohup python3 -u train_sirst_hq_ubuntu.py \
--data_root "$ROOT" --train_txt "${TRAIN_DIR}/train.txt" --val_txt "${TRAIN_DIR}/test.txt" \
--size $SIZE \
--sctransnet_preproc --sc_use_gamma --sc_pos_prob 0.5 \
--batch_size $BS --epochs $EPOCHS --model "$MODEL" \
--boundary_prior_sampling --boundary_ratio 0.5 \
--use_point_loss --point_loss_points 4096 --point_loss_weight 0.3 \
--val_thr_search \
--use_radial_gate_hq --rgate_loc both --freq_patch_size_hq 4 \
--radial_bins_hq 8 --radial_channel_shared_hq --rgate_strength_enc 0.5 --rgate_strength_dec 0.5 \
--rgate_edge_boost 0.7 --rgate_high_freq_thresh 0.5 \
--init_from_baseline weights/efficient_sam_vitt.pt --hq_warmup_epochs 30 \
--exp_name "$EXP">"$LOG_ROOT/train_${TS}.log" 2>&1 &
